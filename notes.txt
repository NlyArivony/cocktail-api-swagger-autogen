# init node
-> npm init -y

# install dependencies when package.json contain the dependencies:
-> npm install

install dependencies manually
-> npm install express swagger-autogen
-> npm install swagger-ui-express
-> npm install sqlite3
-> npm install sequelize

Run the Swagger documentation generator
-> node swagger.js

run app:
-> node index.js
